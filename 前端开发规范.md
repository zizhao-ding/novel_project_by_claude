## 5. 组件开发规范

### 5.1 基础组件开发
```vue
<!-- BaseButton.vue -->
<template>
  <button 
    :class="['base-btn', btnClass]" 
    :disabled="loading || disabled"
    @click="handleClick"
  >
    <el-icon v-if="icon || loading" class="btn-icon">
      <component :is="loading ? 'Loading' : icon" />
    </el-icon>
    <span v-if="$slots.default" class="btn-text">
      <slot />
    </span>
  </button>
</template>

<script setup lang="ts">
import { computed } from 'vue';
import type { PropType } from 'vue';

interface Props {
  type?: 'primary' | 'success' | 'warning' | 'danger' | 'info' | 'text';
  size?: 'large' | 'default' | 'small';
  icon?: string;
  loading?: boolean;
  disabled?: boolean;
  onClick?: (event: MouseEvent) => void;
}

const props = withDefaults(defineProps<Props>(), {
  type: 'default',
  size: 'default',
  loading: false,
  disabled: false,
});

const emits = defineEmits<{
  (e: 'click', event: MouseEvent): void;
}>();

const btnClass = computed(() => [
  `el-button--${props.type}`,
  `el-button--${props.size}`,
  {
    'is-loading': props.loading,
    'is-disabled': props.disabled,
  },
]);

function handleClick(event: MouseEvent) {
  if (!props.loading && !props.disabled) {
    emits('click', event);
  }
}
</script>
```

## 6. Pinia 状态管理规范

### 6.1 Store 定义
```typescript
// stores/novel.ts
import { defineStore } from 'pinia';
import { ref, computed } from 'vue';
import { getNovelList, uploadNovel } from '@/services/novel';

export const useNovelStore = defineStore('novel', () => {
  // State
  const novels = ref<Novel[]>([]);
  const currentNovel = ref<Novel | null>(null);
  const loading = ref(false);
  const error = ref<string | null>(null);

  // Getters (Computed)
  const novelCount = computed(() => novels.value.length);
  const publishedNovels = computed(() => 
    novels.value.filter(novel => novel.status === 'published')
  );

  // Actions
  async function fetchNovels() {
    loading.value = true;
    error.value = null;
    
    try {
      novels.value = await getNovelList();
    } catch (err) {
      error.value = (err as Error).message;
    } finally {
      loading.value = false;
    }
  }

  async function addNovel(novelData: Partial<Novel>) {
    try {
      const newNovel = await uploadNovel(novelData);
      novels.value.unshift(newNovel);
      return newNovel;
    } catch (err) {
      error.value = (err as Error).message;
      throw err;
    }
  }

  return {
    // State
    novels,
    currentNovel,
    loading,
    error,
    
    // Getters
    novelCount,
    publishedNovels,
    
    // Actions
    fetchNovels,
    addNovel,
  };
});
```

## 7. API 服务规范

### 7.1 服务层组织
```typescript
// services/api.ts
import axios, { AxiosInstance } from 'axios';
import { useUserStore } from '@/stores/user';

class ApiService {
  private instance: AxiosInstance;

  constructor() {
    this.instance = axios.create({
      baseURL: import.meta.env.VITE_API_BASE_URL || '/api',
      timeout: 10000,
    });

    this.setupInterceptors();
  }

  private setupInterceptors() {
    // 请求拦截器
    this.instance.interceptors.request.use(
      (config) => {
        const userStore = useUserStore();
        
        if (userStore.token) {
          config.headers.Authorization = `Bearer ${userStore.token}`;
        }
        
        return config;
      },
      (error) => {
        return Promise.reject(error);
      }
    );

    // 响应拦截器
    this.instance.interceptors.response.use(
      (response) => response.data,
      (error) => {
        if (error.response?.status === 401) {
          // 处理未授权错误
          const userStore = useUserStore();
          userStore.logout();
        }
        
        return Promise.reject(error);
      }
    );
  }

  get client() {
    return this.instance;
  }
}

export const apiService = new ApiService();
```

### 7.2 业务服务定义
```typescript
// services/novel.ts
import { apiService } from './api';
import type { Novel, NovelListResponse } from '@/types/novel';

export const novelApi = {
  // 获取小说列表
  getNovelList: (params: {
    page: number;
    pageSize: number;
    search?: string;
  }) => {
    return apiService.client.get<NovelListResponse>('/novels', { params });
  },

  // 获取单个小说
  getNovel: (id: number) => {
    return apiService.client.get<Novel>(`/novels/${id}`);
  },

  // 上传小说
  uploadNovel: (data: FormData) => {
    return apiService.client.post<Novel>('/novels', data, {
      headers: {
        'Content-Type': 'multipart/form-data',
      },
    });
  },

  // 更新小说
  updateNovel: (id: number, data: Partial<Novel>) => {
    return apiService.client.put<Novel>(`/novels/${id}`, data);
  },

  // 删除小说
  deleteNovel: (id: number) => {
    return apiService.client.delete(`/novels/${id}`);
  },
};
```

## 8. 路由规范

### 8.1 路由配置
```typescript
// router/index.ts
import { createRouter, createWebHistory } from 'vue-router';
import { useUserStore } from '@/stores/user';

const routes = [
  {
    path: '/',
    name: 'Home',
    component: () => import('@/views/Home.vue'),
  },
  {
    path: '/novels',
    name: 'NovelList',
    component: () => import('@/views/NovelList.vue'),
    meta: { requiresAuth: true },
  },
  {
    path: '/novels/:id',
    name: 'NovelDetail',
    component: () => import('@/views/NovelDetail.vue'),
    props: true,
  },
  {
    path: '/upload',
    name: 'Upload',
    component: () => import('@/views/Upload.vue'),
    meta: { requiresAuth: true },
  },
];

const router = createRouter({
  history: createWebHistory(),
  routes,
});

// 路由守卫
router.beforeEach((to, from, next) => {
  const userStore = useUserStore();
  const requiresAuth = to.matched.some(record => record.meta.requiresAuth);

  if (requiresAuth && !userStore.isAuthenticated) {
    next('/login');
  } else {
    next();
  }
});

export default router;
```

## 10. 错误处理规范

### 10.1 全局错误处理
```typescript
// plugins/error-handler.ts
import { ElMessage, ElNotification } from 'element-plus';

export function setupErrorHandler(app: App) {
  // 全局错误处理
  app.config.errorHandler = (err, instance, info) => {
    console.error('Global Error:', err);
    console.error('Error Info:', info);
    
    ElMessage.error('系统出现错误，请稍后重试');
  };

  // Promise 错误处理
  window.addEventListener('unhandledrejection', (event) => {
    console.error('Unhandled Promise Rejection:', event.reason);
    ElMessage.error('网络请求失败，请检查网络连接');
  });
};
```

## 13. 通用最佳实践

### 13.1 条件渲染
```vue
<!-- 正确：避免在模板中使用复杂表达式 -->
<template>
  <!-- 不推荐 -->
  <div v-if="user.permissions && user.permissions.includes('admin') && user.isActive">
    Admin Content
  </div>
  
  <!-- 推荐 -->
  <div v-if="isAdminUser">
    Admin Content
  </div>
</template>

<script setup lang="ts">
const isAdminUser = computed(() => 
  user.value?.permissions?.includes('admin') && user.value?.isActive
);
</script>
```

### 13.2 列表渲染
```vue
<!-- 总是为 v-for 提供 key -->
<template>
  <li v-for="item in items" :key="item.id">
    {{ item.name }}
  </li>
</template>
```

### 13.3 性能优化
- 使用 `v-show` 而不是频繁切换的 `v-if`（当元素需要频繁切换显示状态时）
- 对长列表使用虚拟滚动（可引入 vue-virtual-scroller）
- 合理使用 `computed` 缓存复杂计算
- 组件懒加载 `defineAsyncComponent`

这个规范为项目提供了一套完整、标准化的前端开发准则，确保团队成员遵循一致的编码风格和最佳实践，提高代码质量、可维护性和协作效率。
